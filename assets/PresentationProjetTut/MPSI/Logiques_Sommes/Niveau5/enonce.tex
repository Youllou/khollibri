	Soit $n\in \N$. On note  
	\[
		a_n = \sum_{k=1}^n k
		\quad , \quad 
		b_n = \sum_{k=1}^n k^2
		\quad \mbox{ et }\quad 
		c_n = \sum_{k=1}^n k^3 \ .
	\]
	On admet que
		\[
		a_n = \frac{n(n+1)}{2}
		\quad , \quad 
		b_n = \frac{n(n+1)(2n+1)}{6}
		\quad \mbox{ et }\quad 
		c_n = a_n^2 .
	\]
	\begin{enumerate}
		\item Calculer $$\sum_{1\leqslant i \leqslant i \leqslant n} ij.$$		
		\item Calculer $$ \sum_{i=1}^n \sum_{j=1}^n \min(i,j).$$
	\end{enumerate}
